    local OriginalAvatarCache = {
        Cached = false,
        Description = nil,
        UserId = nil
    }

    local function cacheOriginalAvatar()
        if OriginalAvatarCache.Cached then return end
        
        local success, desc = pcall(function()
            return Players:GetHumanoidDescriptionFromUserId(LocalPlayer.UserId)
        end)
        
        if success and desc then
            OriginalAvatarCache.Description = desc
            OriginalAvatarCache.UserId = LocalPlayer.UserId
            OriginalAvatarCache.Cached = true
        end
    end

    -- Cache avatar saat script dimuat
    task.spawn(function()
        task.wait(1)
        cacheOriginalAvatar()
    end)

    local function getPlayerList()
        local list = {}
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer then
                table.insert(list, p.Name)
            end
        end
        table.sort(list)
        return list
    end

    local selectedAvatarPlayer = nil

    local avatarDropdown = AvatarTab:Dropdown({
        Title = "Select Player",
        Desc = "Pilih player yang ingin dicopy avatarnya",
        Values = getPlayerList(),
        Multi = false,
        Callback = function(value)
            selectedAvatarPlayer = value
        end
    })

    local function applyMorphEffect(character)
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if not rootPart then return end

        local particleEmitter = Instance.new("ParticleEmitter")
        particleEmitter.Texture = "rbxassetid://243098098"
        particleEmitter.Rate = 50
        particleEmitter.Speed = NumberRange.new(5, 10)
        particleEmitter.Lifetime = NumberRange.new(0.5, 1)
        particleEmitter.SpreadAngle = Vector2.new(360, 360)
        particleEmitter.Color = ColorSequence.new(Color3.fromRGB(200, 50, 50))
        particleEmitter.Parent = rootPart

        local explosion = Instance.new("Explosion")
        explosion.BlastRadius = 5
        explosion.BlastPressure = 0
        explosion.Position = rootPart.Position
        explosion.Visible = true
        explosion.Parent = workspace
        explosion.ExplosionType = Enum.ExplosionType.NoCraters

        task.spawn(function()
            task.wait(2)
            particleEmitter.Enabled = false
            task.wait(1)
            particleEmitter:Destroy()
            explosion:Destroy()
        end)
    end

    local function clearCurrentAppearance(character)
        -- Hapus semua item appearance yang ada
        for _, obj in ipairs(character:GetChildren()) do
            if obj:IsA("Shirt") or obj:IsA("Pants") or obj:IsA("ShirtGraphic") or
               obj:IsA("Accessory") or obj:IsA("BodyColors") then
                obj:Destroy()
            end
        end
        
        -- Hapus decals di head
        local head = character:FindFirstChild("Head")
        if head then
            for _, decal in ipairs(head:GetChildren()) do
                if decal:IsA("Decal") then 
                    decal:Destroy() 
                end
            end
        end
    end

    local function morphToPlayer(targetName)
        if not targetName then 
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Pilih player terlebih dahulu!",
                Duration = 3,
                Icon = "x"
            })
            return 
        end
        
        local target = Players:FindFirstChild(targetName)
        if not target then
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Player tidak ditemukan!",
                Duration = 3,
                Icon = "x"
            })
            return
        end
        
        local userId = target.UserId
        
        if userId == LocalPlayer.UserId then
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Tidak bisa copy avatar sendiri!",
                Duration = 3,
                Icon = "x"
            })
            return
        end
        
        -- Cache avatar original jika belum
        cacheOriginalAvatar()
        
        local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
        local humanoid = character:WaitForChild("Humanoid", 10)
        if not humanoid then 
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Humanoid tidak ditemukan!",
                Duration = 3,
                Icon = "x"
            })
            return 
        end

        local success, desc = pcall(function()
            return Players:GetHumanoidDescriptionFromUserId(userId)
        end)
        if not success or not desc then
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Gagal memuat data avatar!",
                Duration = 3,
                Icon = "x"
            })
            return
        end

        clearCurrentAppearance(character)

        local applySuccess = pcall(function()
            humanoid:ApplyDescriptionClientServer(desc)
        end)

        if applySuccess then
            applyMorphEffect(character)
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Berhasil copy avatar " .. targetName .. "!",
                Duration = 3,
                Icon = "check"
            })
        else
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Gagal menerapkan avatar!",
                Duration = 3,
                Icon = "x"
            })
        end
    end

    local function resetToOriginalAvatar()
        if not OriginalAvatarCache.Cached or not OriginalAvatarCache.Description then
            WindUI:Notify({
                Title = "Reset Avatar",
                Content = "Avatar original belum di-cache! Tunggu sebentar...",
                Duration = 3,
                Icon = "x"
            })
            
            -- Coba cache ulang
            cacheOriginalAvatar()
            task.wait(1)
            
            if not OriginalAvatarCache.Cached then
                WindUI:Notify({
                    Title = "Reset Avatar",
                    Content = "Gagal memuat avatar original!",
                    Duration = 3,
                    Icon = "x"
                })
                return
            end
        end
        
        local character = LocalPlayer.Character
        if not character then
            WindUI:Notify({
                Title = "Reset Avatar",
                Content = "Character tidak ditemukan!",
                Duration = 3,
                Icon = "x"
            })
            return
        end
        
        local humanoid = character:FindFirstChild("Humanoid")
        if not humanoid then
            WindUI:Notify({
                Title = "Reset Avatar",
                Content = "Humanoid tidak ditemukan!",
                Duration = 3,
                Icon = "x"
            })
            return
        end
        
        clearCurrentAppearance(character)
        
        local success = pcall(function()
            humanoid:ApplyDescriptionClientServer(OriginalAvatarCache.Description)
        end)
        
        if success then
            applyMorphEffect(character)
            WindUI:Notify({
                Title = "Reset Avatar",
                Content = "Avatar berhasil direset ke default!",
                Duration = 3,
                Icon = "check"
            })
        else
            WindUI:Notify({
                Title = "Copy Avatar",
                Content = "Gagal mereset avatar!",
                Duration = 3,
                Icon = "x"
            })
        end
    end

    -- Animation functions
    local function StopAnim()
        pcall(function()
            local char = LocalPlayer.Character
            if not char or not char:FindFirstChild("Animate") then return end
            
            char.Animate.Disabled = false
            local hum = char:FindFirstChildOfClass("Humanoid")
            if not hum then return end
            
            local animtrack = hum:GetPlayingAnimationTracks()
            for i, track in pairs(animtrack) do
                track:Stop()
            end
        end)
    end

    local function cacheOriginalAnimations()
        local char = LocalPlayer.Character
        local Animate = char and char:FindFirstChild("Animate")
        if not Animate or next(originalAnimationIDs) then return end 
        
        pcall(function()
            originalAnimationIDs = {
                idle1 = Animate.idle.Animation1.AnimationId:match("%d+"),
                idle2 = Animate.idle.Animation2.AnimationId:match("%d+"),
                walk = Animate.walk.WalkAnim.AnimationId:match("%d+"),
                run = Animate.run.RunAnim.AnimationId:match("%d+"),
                jump = Animate.jump.JumpAnim.AnimationId:match("%d+"),
                climb = Animate.climb.ClimbAnim.AnimationId:match("%d+"),
                fall = Animate.fall.FallAnim.AnimationId:match("%d+"),
            }
        end)
    end

    local function applyAnimationPack(ids)
        local char = LocalPlayer.Character
        local Animate = char and char:FindFirstChild("Animate")
        if not Animate then 
            WindUI:Notify({
                Title = "Animation",
                Content = "Character 'Animate' script not found.",
                Duration = 3
            })
            return 
        end
        
        cacheOriginalAnimations() 
        
        local animsToApply = {}
        
        if not ids then 
            animsToApply = originalAnimationIDs
            if not next(animsToApply) then
                Animate.Disabled = false
                return
            end
        elseif ids.isPartial then 
            for k, v in pairs(originalAnimationIDs) do animsToApply[k] = v end
            for k, v in pairs(ids) do
                if k ~= "label" and k ~= "isPartial" then
                    animsToApply[k] = v
                end
            end
        else 
            animsToApply = ids
        end
        
        local labelText
        if not ids then
            labelText = "Default"
        elseif ids.label then
            labelText = ids.label
        else
            labelText = "Custom"
        end
        
        pcall(function()
            Animate.Disabled = true
            StopAnim()
            
            local prefix = "http://www.roblox.com/asset/?id="
            if animsToApply.idle1 then Animate.idle.Animation1.AnimationId = prefix .. animsToApply.idle1 end
            if animsToApply.idle2 then Animate.idle.Animation2.AnimationId = prefix .. animsToApply.idle2 end
            if animsToApply.walk then Animate.walk.WalkAnim.AnimationId = prefix .. animsToApply.walk end
            if animsToApply.run then Animate.run.RunAnim.AnimationId = prefix .. animsToApply.run end
            if animsToApply.jump then Animate.jump.JumpAnim.AnimationId = prefix .. animsToApply.jump end
            if animsToApply.climb then Animate.climb.ClimbAnim.AnimationId = prefix .. animsToApply.climb end
            if animsToApply.fall then Animate.fall.FallAnim.AnimationId = prefix .. animsToApply.fall end
            
            Animate.Disabled = false
            WindUI:Notify({
                Title = "Animation",
                Content = "Applied: " .. labelText,
                Duration = 3
            })
        end)
    end

        local currentTab = tabs[currentTabIndex]
        local newTab = tabs[newIndex]
        
        -- Slide out current tab
        TweenService:Create(currentTab, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
            Position = UDim2.new(-1, 0, 0, 0)
        }):Play()
        
        task.wait(0.1)
        
        -- Position new tab for slide in
        newTab.Position = UDim2.new(1, 0, 0, 0)
        newTab.Visible = true
        
        -- Slide in new tab
        TweenService:Create(newTab, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Position = UDim2.new(0, 0, 0, 0)
        }):Play()
        
        task.wait(0.2)
        currentTab.Visible = false
        
        currentTabIndex = newIndex
    end

    -- Hook into tab selection
    local originalSelectTab = Window.SelectTab
    Window.SelectTab = function(self, index)
        if originalSelectTab then
            originalSelectTab(self, index)
        end
        switchTabWithSlide(index)
    end
